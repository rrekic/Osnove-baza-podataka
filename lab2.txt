salary AS plata

2;
SELECT
first_name || ' ' || last_name AS "Naziv zaposlenog",
department_id AS "Sifra odjela"
FROM employees

3;
SELECT
first_name || ' ' || last_name AS "Naziv zaposlenog",
salary AS plata
FROM employees
WHERE salary NOT BETWEEN 1000 AND 2345;

4;
SELECT
first_name || ' ' || last_name AS "Zaposleni",
job_id AS posao
hire_date AS "Datum zaposlenja"
FROM employees
WHERE hire_date BETWEEN To_Date('11-01-1996', 'DD-MM-YYYY') AND To_Date('22-02-1997', 'DD-MM-YYYY');

5;
SELECT
first_name || ' ' || last_name AS "Naziv zaposlenog",
department_id AS "Sifra odjela"
FROM employees
WHERE department_id BETWEEN 10 AND 30
ORDER BY last_name;

6;
SELECT
salary AS "Mjesecna plata",
first_name AS "Ime zaposlenog",
last_name AS "Prezime zaposlenog",
Nvl(commission_pct,0) "Dodatak na platu"
FROM employees
WHERE salary>1500 AND department_id BETWEEN 10 AND 30;

7;
SELECT *
FROM employees
WHERE hire_date < To_Date('01-01-1996', 'DD-MM-YYYY');

8;
SELECT
first_name || ' ' || last_name AS "Naziv zaposlenog",
salary AS Plata,
job_id AS Posao
FROM employees
WHERE manager_id IS NULL;

9;
SELECT
first_name || ' ' || last_name AS "Naziv zaposlenog",
salary AS Plata,
commission_pct AS "Dodatak na platu"
FROM employees
WHERE commission_pct IS NOT NULL
ORDER BY salary DESC, commission_pct DESC;

10;

first_name || ' ' || last_name AS "Naziv zaposlenog"
FROM employees
WHERE first_name || last_name LIKE '%l%l';

11;
SELECT first_name || ' ' || last_name AS Zaposleni,
salary AS Plata,
commission_pct "Dodatak na platu"
FROM employees
WHERE (Nvl(commission_pct, 0) * salary) > (salary - (0.8 * salary));
