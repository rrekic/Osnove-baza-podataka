SELECT
e.first_name || ' ' || e.last_name AS "Naziv zaposlenog",
j.job_title AS Posao,
d.department_name AS Odjel,
e.salary AS Plata,
j.min_salary, j.max_salary
FROM employees e, departments d, jobs j
WHERE e.job_id=j.job_id AND e.department_id=d.department_id AND
e.salary * (1 + Nvl(e.commission_pct, 0)) NOT BETWEEN j.min_salary AND j.max_salary;
